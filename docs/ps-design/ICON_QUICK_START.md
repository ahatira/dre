```javascript
// ADD THIS FUNCTION (after buildIcons function)

/**
 * Generate icons.css with ALL icon mappings
 * Input: Array of icon names
 * Output: CSS file with [data-icon="*"] rules
 */
function generateIconsCss(iconNames) {
  const cssLines = [
    '/**',
    ' * GENERATED FILE - DO NOT EDIT MANUALLY',
    ' * Generated by: scripts/build-icons.mjs',
    ' * Generated: ' + new Date().toISOString(),
    ' *',
    ' * To add icons: Drop SVG in source/icons-source/, run: npm run build',
    ' */',
    '',
    '/* Base styling for all data-icon elements */',
    '[data-icon] {',
    '  display: inline-block;',
    '  width: 1em;',
    '  height: 1em;',
    '  font-style: normal;',
    '  font-variant: normal;',
    '  text-transform: none;',
    '  line-height: 1;',
    '  -webkit-font-smoothing: antialiased;',
    '  -moz-osx-font-smoothing: grayscale;',
    '  background-repeat: no-repeat;',
    '  background-position: center;',
    '  background-size: contain;',
    '}',
    '',
    '/* Icon mappings (auto-generated - ' + iconNames.length + ' icons) */',
  ];

  // Generate one rule per icon
  for (const name of iconNames.sort()) {
    cssLines.push(
      `[data-icon="${name}"] { background-image: url('/icons/icons-sprite.svg#icon-${name}'); }`
    );
  }

  return cssLines.join('\n') + '\n';
}

/**
 * Generate icons-registry.json for validation
 * Input: Array of icon names
 * Output: JSON registry with metadata
 */
function generateIconsRegistry(iconNames) {
  const registry = {
    generated: new Date().toISOString(),
    total: iconNames.length,
    names: iconNames.sort(),
    categories: {
      // Organize by prefix (optional, can be enhanced)
      ui: iconNames.filter(n => ['close', 'check', 'info', 'alert', 'help'].includes(n)),
      navigation: iconNames.filter(n => n.includes('arrow') || n.includes('chevron')),
      forms: iconNames.filter(n => n.includes('checkbox') || n.includes('radio')),
    }
  };
  return JSON.stringify(registry, null, 2) + '\n';
}

// MODIFY: buildIcons() function to call new generators

async function buildIcons() {
  const svgFiles = await glob('*.svg', {
    cwd: ICONS_SOURCE_DIR,
  });

  if (!svgFiles.length) {
    throw new Error(`No SVG files found in ${ICONS_SOURCE_DIR}`);
  }

  const symbols = [];
  const iconNames = [];  // ADD THIS

  for (const file of svgFiles.sort()) {
    const name = path.basename(file, '.svg');
    const raw = fs.readFileSync(path.join(ICONS_SOURCE_DIR, file), 'utf8');

    const viewBox = VIEWBOX_RE.exec(raw)?.[1] ?? '0 0 24 24';
    const bodyMatch = SVG_SYMBOL_RE.exec(raw);
    let body = bodyMatch ? bodyMatch[1].trim() : raw.trim();

    body = stripSVGFillStroke(body);

    iconNames.push(name);  // ADD THIS
    symbols.push(`<symbol id="icon-${name}" viewBox="${viewBox}">${body}</symbol>`);
  }

  // EXISTING: Write sprite
  const spriteContent = [
    '<svg xmlns="http://www.w3.org/2000/svg" aria-hidden="true" style="position:absolute;width:0;height:0;overflow:hidden">',
    symbols.join('\n'),
    '</svg>',
    '',
  ].join('\n');

  // ... existing sprite write code ...

  // ADD THIS: Generate icons.css
  const cssContent = generateIconsCss(iconNames);
  const cssPath = path.join(__dirname, '../source/props/icons-generated.css');
  fs.mkdirSync(path.dirname(cssPath), { recursive: true });
  fs.writeFileSync(cssPath, cssContent, 'utf8');
  console.log(`✔ Generated CSS (${iconNames.length} rules) → ${cssPath}`);

  // ADD THIS: Generate registry
  const registryContent = generateIconsRegistry(iconNames);
  const registryPath = path.join(__dirname, '../source/patterns/documentation/icons-registry.json');
  fs.mkdirSync(path.dirname(registryPath), { recursive: true });
  fs.writeFileSync(registryPath, registryContent, 'utf8');
  console.log(`✔ Generated registry (${iconNames.length} icons) → ${registryPath}`);
}
```
